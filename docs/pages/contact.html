Contact form

viewof contactForm = Inputs.form({
  Name: Inputs.text({ placeholder: "Your name" }),
  Email: Inputs.text({ placeholder: "your.email@example.com", type: "email" }),
  Subject: Inputs.select(
    ["General Inquiry", "Data Correction", "Collaboration", "Technical Support", "Other"],
    { value: "General Inquiry" }
  ),
  Message: Inputs.textarea({ placeholder: "Your message here...", rows: 6 })
})
viewof submit = Inputs.button("Send Message", { value: 0 })
submissionResult = {
  if (contactForm && contactForm.Name && contactForm.Email && contactForm.Message) {
    return html`
      <div class="form-success">
        <strong>Thank you for your message.</strong><br>
        We'll get back to <b>${contactForm.Email}</b> as soon as possible.
        <br><br>
        <em>This is still a demo form. In production, it should connect to an email service or backend endpoint.</em>
      </div>
    `
  }
  return html``
}
