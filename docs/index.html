Filters

viewof selectedRegion = Inputs.select(regions, {
  label: "Region",
  value: "All Regions"
})

viewof selectedRiskType = Inputs.select(riskTypes, {
  label: "Risk Type",
  value: "All Types"
})

viewof minRiskScore = Inputs.range([0, 10], {
  label: "Minimum Risk Score",
  value: 0,
  step: 0.5
})
// Filter data based on user selections
filteredData = data.filter(d => {
  const regionMatch = selectedRegion === "All Regions" || d.region === selectedRegion
  const riskMatch = selectedRiskType === "All Types" || d.risk_type === selectedRiskType
  const scoreMatch = d.risk_score >= minRiskScore
  return regionMatch && riskMatch && scoreMatch
})

// Leaflet CSS
html`<link rel="stylesheet" href="https://unpkg.com/leaflet@1.9.4/dist/leaflet.css"/>`
leaflet = require("leaflet@1.9.4")
mapDisplay = {
  const container = html`<div class="map-container"><div class="map-inner"></div></div>`
  yield container

  // Initialize map on the inner element
  const map = leaflet.map(container.querySelector(".map-inner")).setView([20, 0], 2)

  leaflet.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
    attribution: "© OpenStreetMap contributors",
    maxZoom: 18
  }).addTo(map)

  const colors = {
    Flood: "#3498db",
    Drought: "#f39c12",
    Wildfire: "#e74c3c",
    Storm: "#9b59b6"
  }

  filteredData.forEach(d => {
    leaflet.circleMarker([d.lat, d.lon], {
      radius: d.risk_score * 2.5,
      fillColor: colors[d.risk_type] ?? "#64748b",
      color: "#fff",
      weight: 2,
      fillOpacity: 0.7
    })
      .bindPopup(`
        <b>${d.country}</b><br>
        Type: ${d.risk_type}<br>
        Score: ${d.risk_score}/10<br>
        Affected: ${d3.format(",")(d.population_affected)}
      `)
      .addTo(map)
  })

  setTimeout(() => map.invalidateSize(), 100)
}
